PROCEDURE DBDelete

*-----------------------------------------------------------------------

*-- Programmer..: Romain Strieff (CIS: 73612,2257)

*-- Date........: 08/10/1994

*-- Notes.......: This procedure will allow wild card deleting with

*--               dBase code alone.

*-- Written for.: dBASE IV, 2.0, 1.5 (1.1 too if pcount()is replaced)

*-- Rev. History: 1994 ?-- Original posted to me by Kenneth (Zak) Chan

*--               06/03/1994 -- eliminated error when sub-dir empty

*--               07/20/1994 -- eliminated error leaving 1 file undel.

*--               07/25/1994 -- added wild card parameter

*-- Calls.......: None

*-- Called by...: Any

*-- Usage.......: do dbdelete with cDirSet,cWildCard

*-- Example.....: do dbdelete with "c:\data","*.??k"

*-- Returns.....: Nil

*-- Parameters..: cDirSet   -- directory to delete (c:\data, a: etc)

*--               cWildCard -- files to delete (*.txt *.??k etc)

*-- WARNING.....: If cWildCard is omitted, procedure will delete 

*--               ALL files !!!!

*-----------------------------------------------------------------------


   parameters cDirSet,cWildCard

   private cConsole,cSafety,cOldDir,nbar


   *-- check for blank parameter

   if isblank(m->cDirSet)

      return

   endif && len(trim(m->cDirSet))=0


   if right(trim(m->cDirSet),1)="\"

      cDirSet=left(m->cDirSet,len(m->cDirSet)-1)

   endif && right(trim(m->cDirSet),1)="\"


   cWildCard=iif(pcount()<2,"*.*",m->cWildCard)


   cSafety=set("SAFETY")

   cConsole=set("CONSOLE")

   set console off

   set safety off

   cOldDir=set("DIRECTORY")

   set directory to &cDirSet. && a:\ or c:\temp\ or whatever


   *-- Get all the file names

   define popup PDelFiles Prompt Files like &cWildCard. from 0,0

   *-- File names start at sixth prompt

   if barcount("PDELFILES")>5

      nBar = 6

      *-- Skip all the subdirectories

      do while "<" $ barprompt( m->nBar, "pDelFiles" )

         m->nBar = m->nBar + 1

      enddo && while "<" $ barprompt( m->nBar, "pDelFiles" )

      *-- Delete all the files in the subdirectory

      do while barcount( "pDelFiles" ) >= m->nBar

         erase ( barprompt( m->nBar, "pDelFiles" ))

      enddo && while barcount( "pDelFiles" ) >= m->nBar

   endif && barcount("PDELFILES")>5

   release popup PDelFiles

   set dire to &cOldDir.

   if m->cConsole="ON"

      set console on

   endif && cConsole="ON"

   if m->cSafety="ON"

      set safety on

   endif && cSafety="ON"


RETURN

*-- EoP: DBDelete
