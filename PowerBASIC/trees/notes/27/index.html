FUNCTION conout(txt$,OPT ByVal arg as DWORD) as DWORD

  ' -------------------------------------------
  ' conout basic$       = default appends CRLF
  ' conout basic$, 0    = default appends CRLF
  ' conout basic$, 1    = non zero omits CRLF
  ' bytes_written = conout(basic$,[opt arg])
  ' -------------------------------------------

    #REGISTER NONE

    LOCAL hOutPut  as DWORD
    LOCAL bWritten as DWORD
    LOCAL slen     as DWORD
    LOCAL ptxt     as DWORD

    If arg = 0 Then
      txt$ = txt$ + $CRLF
    End if

    ptxt = StrPtr(txt$)

    PREFIX "! "
    mov eax, ptxt           ; load address into EAX
    mov eax, [eax-4]        ; get length store 4 bytes below
    mov slen, eax           ; write it to a variable
    END PREFIX

    hOutPut = CO_GetStdHandle(CO_STD_OUTPUT_HANDLE)
    CO_WriteFile hOutPut,ptxt,slen,VarPtr(bWritten),0

    FUNCTION = bWritten

END FUNCTION

                and by the way the function works just fine.
                It does, but as said, a std handle needs to be set up in the first place for it to. Which needs either an AttachConsole, or a SetStdHandle as by default, GUI apps have null ones.

                By way of alternative, you can also set it up to use the crt functions. You have to use the f versions if you want to bother with printf/scanf et al since the C standard handles llike stdin, stdout and stderr need freopen-ing to actually be attached to something.

                Code:
                #compile exe
#Include once "WIN32API.INC"

DECLARE FUNCTION fputs CDECL IMPORT "msvcrt.dll" ALIAS "fputs" ( _
    BYREF string AS STRINGZ, _
    BYVAL stream AS DWORD _
 ) AS LONG

DECLARE FUNCTION fopen CDECL IMPORT "msvcrt.dll" ALIAS "fopen" ( _
    BYREF filename AS STRINGZ, _
    BYREF mode AS STRINGZ _
 ) AS DWORD

DECLARE FUNCTION fflush CDECL IMPORT "msvcrt.dll" ALIAS "fflush" ( _
    BYVAL fileStream AS DWORD _
) AS LONG

GLOBAL g_stdout AS DWORD

SUB PrintString(BYREF str AS STRINGZ)
    fputs(str, g_stdout)
    fflush(g_stdout)
END SUB

FUNCTION PBMain()

    ' Fake being a console program
    IF AttachConsole(-1) = 0 THEN
        ? "Wasn't run from a console!"
        EXIT FUNCTION
    END IF

    g_stdout = fopen("CONOUT$", "wt")

    PrintString("Hi from console")

END FUNCTION
