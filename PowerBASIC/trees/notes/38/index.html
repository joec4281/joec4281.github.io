  CreateMMF proc lpName:DWORD,bcnt:DWORD,lphndl:DWORD,lpMemFile:DWORD
  1. lpName The zero terminates string that names the file
  2. bcnt   The size of the required memory mapped file in BYTES.
  3. lphndl Pointer to DWORD variable to receive the address of the memory mapped file handle.
  4. lpMem  Pointer toDWORD variable to receive the start address of the memory mapped file.

  The lphndl and lpMemFile have the file handle and start address written to them when the
  procedure returns. There is no return value.

FUNCTION CreateMMF( _
                    ByVal lpName as DWORD, _
                    ByVal bcnt as DWORD, _
                    ByVal lphndl as DWORD, _
                    ByVal lpMem as DWORD _
                  ) as DWORD

    LOCAL hMMF as DWORD

    hMMF = CreateFileMapping(&HFFFFFFFF, _      ' open file handle
                  ByVal %NULL, _                ' unused security attribute
                  %PAGE_READWRITE, _            ' read write access to memory
                  0, _                          ' unused high DWORD
                  bcnt, _                       ' required size in BYTEs
                  ByVal lpName)                 ' set file object name here
    PREFIX "!"
      mov ecx, lphndl                           ' store address of variable in ECX
      mov [ecx], eax                            ' write MMF handle to variable
    END PREFIX

    MapViewOfFile(hMMF, _
                  %FILE_MAP_COPY, _
                  0,0,0)
    PREFIX "!"
      mov ecx, lpMem                            ' store address of variable in ECX
      mov [ecx], eax                            ' write start address to variable
    END PREFIX

END FUNCTION

END FUNCTION

                      Hi Anne,

                      Many ways to do it... Check for those keywords: InterProcess, ClipBoard, MailSlot, FileMapping, wmCopyData, Pipe, UDP, TCP, etc...

                      On way I like is to use RegisterWindowMessage to get and set handles between two programs.
                      Then WM_COPYDATA can be use to set and get string data from one application to another.

                      Here is a piece of code that do this.
                      Run the exe twice, set different text and see the results.
                      At first, it could be difficult to figure out the way each message is taking,
                      but once it's done, you are in business.

                      As usual, I didn't put many comments, just ask if something is unclear.

                      Pierre


                      Code:
                      #COMPILE EXE ''#Win 9.07#
#DIM ALL
#INCLUDE "Win32Api.Inc"

%Edit       = 101
%ButtonSend = 201
%ButtonPump = 202
%ButtonExit = 203

%HandleSet  = 1
%PumpAction = 2
%QuitAction = 3
$Caption    = "WM_COPYDATA"

GLOBAL hDlg AS DWORD
'_____________________________________________________________________________

CALLBACK FUNCTION DlgProc
 STATIC pCopyData       AS COPYDATASTRUCT POINTER
 STATIC CopyData        AS COPYDATASTRUCT
 STATIC pz              AS ASCIIZ POINTER
 STATIC sData           AS STRING
 STATIC hSibbling       AS DWORD
 LOCAL  hSendingWindow  AS DWORD
 STATIC WM_InfoTransmit AS DWORD

 SELECT CASE CBMSG

   CASE %WM_INITDIALOG
     pCopyData = VARPTR(CopyData) 'A COPYDATASTRUCT for WM_COPYDATA

     WM_InfoTransmit = RegisterWindowMessage("WM_InfoTransmit") 'Our own private Windows message
     PostMessage(%HWND_BROADCAST, WM_InfoTransmit, hDlg, %HandleSet) 'See if another instance is alive

   CASE WM_InfoTransmit
     SELECT CASE CBLPARAM

       CASE %HandleSet                   'Initial sibling hanle get
         hSendingWindow = CBWPARAM       'Message sender handle
         IF hSendingWindow <> hDlg THEN  'Discard our own message
           hSibbling = hSendingWindow    'If not self then it's sibbling
           PostMessage(%HWND_BROADCAST, WM_InfoTransmit, hDlg, %HandleSet) 'Send our handle to the other program
           CONTROL ENABLE hDlg, %ButtonSend 'With both programs having both handles
           CONTROL ENABLE hDlg, %ButtonPump 'those buttons can be used
         END IF

       CASE %QuitAction
         hSibbling = 0                     'Reset sibbling handle
         CONTROL DISABLE hDlg, %ButtonSend 'Other programs ends so
         CONTROL DISABLE hDlg, %ButtonPump 'those buttons can't be used

       CASE %PumpAction
         hSendingWindow = CBWPARAM
         IF hSendingWindow <> hDlg THEN 'Other program asked for data, so press the send button
           PostMessage(hDlg, %WM_COMMAND, MAKDWD(%ButtonSend, %BN_CLICKED), GetDlgItem(hDlg, %ButtonSend))
         END IF
     END SELECT

   CASE %WM_COPYDATA                      'We are receiving or sending data
     hSendingWindow = CBWPARAM            'Handle of sending window
     IF hSendingWindow = hSibbling THEN   'Bypass our own messages
       pCopyData      = CBLPARAM          'CopyData structure pointer is in CBLPARAM
       pz             = @pCopyData.lpData 'Pointer to string data
       SetDlgItemText(hDlg, %Edit, @pz)   'Fill edit control
     END IF

  CASE %WM_COMMAND
    SELECT CASE LOWRD(CBWPARAM)

      CASE %ButtonSend
        IF (CBCTLMSG = %BN_CLICKED) OR (CBCTLMSG = 1) THEN
          sData = NUL$(1 + SendDlgItemMessage(hDlg, %Edit, %WM_GETTEXTLENGTH, 0, 0)) 'Set a buffer to get edit control text
          IF LEN(sData) THEN 'Do we have something
            GetDlgItemText(hDlg, %Edit, BYVAL STRPTR(sData), LEN(sData)) 'Get text
            CopyData.lpData  = STRPTR(sData) 'Set WM_COPYDATA structure
            CopyData.cbdata  = LEN(sData)    'Data lenght
            CopyData.dwData  = 12345 'Use this DWORD to do whatever you want
            SendMessage(hSibbling, %WM_COPYDATA, hDlg, VARPTR(CopyData)) 'Send data to other instance
          END IF
        END IF

      CASE %ButtonPump
        IF (CBCTLMSG = %BN_CLICKED) OR (CBCTLMSG = 1) THEN
          PostMessage(%HWND_BROADCAST, WM_InfoTransmit, hDlg, %PumpAction) 'Get data from other instance
        END IF

      CASE %ButtonExit
        IF (CBCTLMSG = %BN_CLICKED) OR (CBCTLMSG = 1) THEN
          DIALOG END CBHNDL, 0
        END IF

    END SELECT

   CASE %WM_DESTROY
     PostMessage(%HWND_BROADCAST, WM_InfoTransmit, 0, %QuitAction) 'Tell other instance that we are ending

 END SELECT

END FUNCTION
'_____________________________________________________________________________

FUNCTION PBMAIN

 LOCAL Ofset AS LONG : IF FindWindow("", $Caption) THEN Ofset = 210

 DIALOG NEW %HWND_DESKTOP, $Caption, 210 + Ofset, 10, 190, 214, _
 %WS_POPUP OR %DS_MODALFRAME OR %WS_CAPTION OR %WS_MINIMIZEBOX OR %WS_SYSMENU, 0 TO hDlg

 CONTROL ADD TEXTBOX, hdlg, %Edit, "123", 5, 5, 180, 190, %WS_VSCROLL OR _
 %ES_MULTILINE OR %ES_WANTRETURN OR %WS_TABSTOP, %WS_EX_CLIENTEDGE

 CONTROL ADD BUTTON, hDlg, %ButtonSend, "&Send", 40, 197, 35, 15
 CONTROL DISABLE hDlg, %ButtonSend

 CONTROL ADD BUTTON, hdlg, %ButtonPump, "&Pump", 80, 197, 35, 15
 CONTROL DISABLE hDlg, %ButtonPump

 CONTROL ADD BUTTON, hDlg, %ButtonExit, "E&xit", 120, 197, 35, 15

 DIALOG SHOW MODAL hDlg CALL DlgProc

END FUNCTION
'_____________________________________________________________________________
'

                                    Can also use a DLL. Maybe not necessary with only two exes, but useful if several need to share same data. 3 codes follow, 1 DLL, 2 for Prog1 and Prog2:
                                    Code:
                                    '====================================================================
' myDll.bas
'--------------------------------------------------------------------
#COMPILE DLL
#DIM ALL
#INCLUDE "WIN32API.INC"
'--------------------------------------------------------------------
%SHMEMSIZE = 3000000 ' 2 MB + room for extra, change to own liking
GLOBAL pzString AS ASCIIZ PTR

' In all calling exes
'DECLARE SUB SetSharedString LIB "myDll.dll" _
'     ALIAS "SetSharedString" (BYVAL sText AS STRING)

'DECLARE FUNCTION GetSharedString LIB "myDll.dll" _
'          ALIAS "GetSharedString" () AS STRING


'====================================================================
' DLL main..
'--------------------------------------------------------------------
FUNCTION LIBMAIN (BYVAL hInstance   AS DWORD, _
                  BYVAL fwdReason   AS LONG, _
                  BYVAL lpvReserved AS LONG) AS LONG

  LOCAL hMapObj AS DWORD

  SELECT CASE fwdReason
  CASE %DLL_PROCESS_ATTACH
      hMapObj = CreateFileMapping(BYVAL -1, BYVAL %Null, _
                                  BYVAL %PAGE_READWRITE, _
                                  BYVAL 0, %SHMEMSIZE, "myDllMemFilemap")
      IF hMapObj THEN
        pzString = MapViewOfFile (hMapObj, BYVAL %FILE_MAP_ALL_ACCESS, 0, 0, 0)
        IF pzString THEN FUNCTION = 1
      END IF

  CASE %DLL_PROCESS_DETACH
      IF pzString THEN UnMapViewOfFile(pzString)
      IF hMapObj  THEN CloseHandle(hMapObj)
      FUNCTION = 1

  CASE %DLL_THREAD_ATTACH  : FUNCTION = 1
  CASE %DLL_THREAD_DETACH  : FUNCTION = 1
  END SELECT
END FUNCTION

'====================================================================
' Set string for further processing
'--------------------------------------------------------------------
SUB SetSharedString ALIAS "SetSharedString" (BYVAL sText AS STRING) EXPORT
  @pzString = sText '@pzString is now a max %SHMEMSIZE byte global string
END SUB

'====================================================================
' Return processed string
'--------------------------------------------------------------------
FUNCTION GetSharedString ALIAS "GetSharedString" () EXPORT AS STRING
  FUNCTION = @pzString
END FUNCTION
