    PBCC Get Command Line Arguments

    5 Mar 2022, 04:41 AM
    Here is a function to parse the command line, taking account for spaces in file names enclosed in double quotes.
    To be saved as "GetCommandArgs.inc"

    Code:
    '------------------------------------------------------------------------------
'   GetCommandArgs
'   --------------
'
'   passed  : command$ with space-delimited arguments,
'             spaces will remain in double-quoted arguments
'
'   returns : function value # of argument elements in array
'           : filled argument array
'
'------------------------------------------------------------------------------
function GetCommandArgs(sArgs() as string) as long
    local lTemp as long
    local sTemp as string
    sTemp = ReplaceChar(command$,$spc,",")
    lTemp = parsecount(sTemp)
    redim sArgs(1 to lTemp)
    parse sTemp, sArgs()
    function = lTemp
end function

'------------------------------------------------------------------------------
'   Replace Characters / unquoted strings only
'   ------------------------------------------
'
'   passed  : original string
'           : old character
'           : new character
'
'   returns : function value with processed string
'
'------------------------------------------------------------------------------
function ReplaceChar(sData as string, _
                     sOld as string*1, _
                     sNew as string*1) as string
    local lSwDQ as long
    local lTemp as long
    local sTemp as string
    lSwDQ = not lSwDQ
    sTemp = sData
    for lTemp = 1 to len(sTemp)
        if mid$(sTemp,lTemp,1) = $dq then
           lSwDQ = not lSwDQ
        end if
        if mid$(sTemp,lTemp,1) = sOld and lSwDQ then
           mid$(sTemp,lTemp,1) = sNew
        end if
    next lTemp
    function = sTemp
end function
    The following is a tiny program to test GetCommandArgs:

    Code:
    '  ---------------------------------------------------------------------------
'  Source name     : test.bas
'  Executable name : test.exe

'  Created date    : 04-Mar-2022

function pbmain () as long
    dim   sArgs() as string
    local lTemp   as long

    lTemp = GetCommandArgs(sArgs())

    stdout format$(lTemp)+" arguments:"
    for lTemp = 1 to ubound(sArgs)
        stdout sArgs(lTemp)
    next lTemp

end function

    #COMPILE EXE

FUNCTION PBMAIN () AS LONG
     LOCAL lArgs AS LONG
     IF COMMAND$ > "" THEN
         CON.PRINT "Full command line: ", COMMAND$
         lArgs = 1
         WHILE COMMAND$(lArgs) > ""
             CON.PRINT "Argument ";lArgs;":  ";COMMAND$(lArgs)
            INCR lArgs
        WEND
        DECR lArgs
        CON.PRINT lArgs;  " command line arguments"
     ELSE
         CON.PRINT "No command line arguments
     END IF
     CON.PRINT "... any key to quit"
     WAITKEY$
END FUNCTION

    And here is the simplified (and improved) version of GetCommandArgs many thanks to Stuart McLachlan!
    Code:
    '------------------------------------------------------------------------------
'   GetCommandArgs
'   --------------
'
'   passed  : command$ with space-delimited arguments,
'             spaces will remain in double-quoted arguments
'
'   returns : function value # of argument elements in array
'           : filled argument array
'
'------------------------------------------------------------------------------
function GetCommandArgs(sArgs() as string) as long
    local lTemp1 as long
    local lTemp2 as long
    lTemp1 = 1
    while len(command$(lTemp1))
      incr lTemp1
    wend
    decr lTemp1
    if lTemp1 then
       redim sArgs(1 to lTemp1)
       for lTemp2 = 1 to lTemp1
           sArgs(lTemp2) = command$(lTemp2)
       next lTemp2
    end if
    function = lTemp1
end function

    Process switches rather than positional arguments. This thread has no comments link, but this is the same subject.
    Code:
    FUNCTION PBMAIN () AS LONG

 LOCAL sCommand    AS STRING
 LOCAL sDelimiter  AS STRING
 LOCAL sResult()   AS STRING
 LOCAL x           AS LONG

 CON.CAPTION$ = "Switches A to Z into elements of an array"

 sCommand   = "/WHow  /Qnow  /Mbrown /Acow"
 sDelimiter = "/"
 ParseCommandLine sCommand$,sDelimiter$,sResult$()

 ? "ELEMENT","SWITCH","DATA"
 FOR x = 1 TO UBOUND(sResult())
  ? x,LEFT$(sResult(x),2),WRAP$(MID$(sResult$(x),3),$DQ,$DQ)
 NEXT

 ?:? "Press a key"WAITKEY$

END FUNCTION

FUNCTION ParseCommandLine(sCommand AS STRING,sDelimiter AS STRING,sResult() AS STRING) AS LONG
 LOCAL s       AS STRING
 LOCAL istart  AS LONG
 LOCAL iend    AS LONG
 LOCAL counter AS LONG
 DIM sResult(1 TO 26) AS STRING 'account for /A to /Z

 DO
  istart = INSTR(istart+1,sCommand,sDelimiter)
  IF istart = 0 THEN EXIT DO            'done

  INCR counter                         'element counter
  iend = INSTR(istart+1,sCommand,sDelimiter)  'next switch
  IF iend THEN                         '  yes, get text
   sResult(counter) = MID$(sCommand,istart,iend-istart)
  ELSE                                 '  no next switch
   sResult(counter) = MID$(sCommand,istart) 'get text
   EXIT DO                             'done
  END IF
 LOOP
 IF counter = 0 THEN ERASE sResult ELSE REDIM PRESERVE sResult(1 TO counter)
 FUNCTION = counter
END FUNCTION
